-- Fun√ß√£o para criar bot√£o
local function createButton(parent, text, color, callback)
    local button = Instance.new("TextButton", parent)
    button.Size = UDim2.new(0, 230, 0, 50)
    button.Text = text
    button.Font = Enum.Font.SourceSansBold
    button.TextSize = 16
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.BackgroundColor3 = color
    button.BorderSizePixel = 0 -- Sem bordas

    -- Conectar a fun√ß√£o ao bot√£o
    button.MouseButton1Click:Connect(function()
        if type(callback) == "function" then
            callback()
        else
            warn("Callback fornecido n√£o √© uma fun√ß√£o!")
        end
    end)

    return button
end

-- Fun√ß√µes de Treinamento Autom√°tico

    _G.fastTrain = true
    while _G.fastTrain do
        local args = { [1] = "rep" } -- Evento de treino
        game:GetService("Players").LocalPlayer.muscleEvent:FireServer(unpack(args))
        wait(0.05) -- Treino r√°pido
    end
end

local function disableFastTrain()
    _G.fastTrain = false
    print("Treino r√°pido desativado!")
end

    while _G.autoRebirth do
        local currentStrength = player.leaderstats.Strength.Value
        local requiredStrength = (player.leaderstats.Rebirths.Value + 1) * 5000
